# WebpackChainer

A function that will receive the group of webpack-chain configurations instance.

type WebpackChainer = (webpackConfigs: WebpackChainConfigGroup) => void;

# webpackConfigs

Type: WebpackChainConfigGroup

Group of all webpack-chain's configurations instance that's created in vue-pack.

# WebpackChainConfigGroup

Group of all webpack-chain's configurations instance.

interface WebpackChainConfigGroup {
	readonly commonJSConfig;
	readonly esModuleConfig;
	readonly ssrConfig;
	readonly webUnminConfig;
	readonly webMinConfig;
}
